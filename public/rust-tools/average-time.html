Average Time

Takes a function or closure and a number of times to run the function, and returns the average and standard deviation runtime.

pub fn average(f: F, iter: u64) -> (u64, u32)
where
  F: Fn(),
{
  let mut times: Vec = vec![iter; 0];
  for _ in 0..iter {
    let t = time(&f);
    times.push(t);
  }
