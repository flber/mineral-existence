Fast Zip

Efficiently zips two Vec<&str>'s together and returns an owned String. If the vec's don't match in length, the remainder of the longer list is added to the end.

Args:
xs: A Vec<&str>. Zipped together with ys, xs; xs goes first.
ys: A Vec<&str>. ys is zipped second.
Returns:
An owned string made by zipping xs and ys.

pub fn fast_zip(xs: Vec<&str>, ys: Vec<&str>) -> String {
  let mut extra = "";
  if xs.len() != ys.len() {
    if xs.len() > ys.len() {
      extra = xs[xs.len() - 1];
    } else {
      extra = ys[ys.len() - 1];
    }
  }

  let mut zipped = xs
    .iter()
    .zip(ys.iter())
    .map(|(a, b)| vec![*a, *b].concat())
    .collect::>();
  zipped.push(String::from(extra));

  zipped.join("")
}
